# Пояснения по лабораторной работе №13

Эта работа открывает цикл работ по одномерным массивам.

Первоначальная задача кажется преподавателям слишком лёгкой, поэтому вводим дополнительное условие: необходимо предусмотреть ручной и автоматический ввод каждого из массивов.

Итак, когда все условия понятны, идём кодить. Разберём код решения:
```c
#include <stdio.h> // Подключение основной библиотеки в языке для ввода/вывода
#include <stdlib.h> // Подключение модуля для работы с функциями общего назначения, в т.ч. случайных чисел
#include <time.h> // Подключение модуля для работы с системным временем (необходимо для генерации случайных чисел)
#include <windows.h> // Подключение модуля windows.h для установки кодировки вывода

int main() { // Объявляем основную функцию
    SetConsoleOutputCP(CP_UTF8); // Устанавливаем кодировку UTF-8 для вывода в консоли русских символов

    int n, choice; // Объявляем переменные для размера массивов и выбора способа ввода
    float A, productSum = 0; // Объявляем переменную A и инициализируем переменную для суммирования произведений
    int count = 0; // Инициализируем счётчик для подсчета количества произведений

    printf("Введите значение A: "); // Выводим запрос на ввод значения A
    scanf("%f", &A); // Читаем значение A

    printf("Введите размер массивов X и Y: "); // Запрашиваем размер массивов X и Y
    scanf("%d", &n); // Читаем размер массивов

    float X[n], Y[n]; // Объявляем массивы X и Y размером n

    printf("Выберите способ ввода массива X (1 - ручной, 2 - автоматический): "); // Запрашиваем способ ввода для массива X
    scanf("%d", &choice); // Читаем выбор способа ввода

    if (choice == 1) { // Если выбран ручной ввод
        printf("Введите элементы массива X:\n"); // Выводим приглашение для ввода элементов массива X
        for (int i = 0; i < n; i++) { // Цикл для ввода элементов массива X
            printf("X[%d] = ", i); // Выводим индекс элемента массива для ввода
            scanf("%f", &X[i]); // Читаем значение элемента массива
        }
    } else if (choice == 2) { // Если выбран автоматический ввод
        srand(time(NULL)); // Инициализируем генератор случайных чисел текущим временем
        printf("Сгенерированный массив X:\n"); // Выводим информацию о сгенерированном массиве X
        for (int i = 0; i < n; i++) { // Цикл для генерации и вывода элементов массива X
            X[i] = (float)(rand() % 100) / 10; // Генерируем случайное число для элемента массива
            printf("X[%d] = %.2f\n", i, X[i]); // Выводим сгенерированное значение элемента массива
        }
    }

    printf("Выберите способ ввода массива Y (1 - ручной, 2 - автоматический): "); // Аналогично запрашиваем способ ввода для массива Y
    scanf("%d", &choice); // Читаем выбор способа ввода

    if (choice == 1) { // Если выбран ручной ввод для массива Y
        printf("Введите элементы массива Y:\n"); // Выводим приглашение для ввода элементов массива Y
        for (int i = 0; i < n; i++) { // Цикл для ввода элементов массива Y
            printf("Y[%d] = ", i); // Выводим индекс элемента массива для ввода
            scanf("%f", &Y[i]); // Читаем значение элемента массива
        }
    } else if (choice == 2) { // Если выбран автоматический ввод для массива Y
        srand(time(NULL)); // Инициализируем генератор случайных чисел текущим временем
        printf("Сгенерированный массив Y:\n"); // Выводим информацию о сгенерированном массиве Y
        for (int i = 0; i < n; i++) { // Цикл для генерации и вывода элементов массива Y
            Y[i] = (float)(rand() % 100) / 10; // Генерируем случайное число для элемента массива
            printf("Y[%d] = %.2f\n", i, Y[i]); // Выводим сгенерированное значение элемента массива
        }
    }

    for (int i = 0; i < n; i++) { // Цикл для обработки элементов массивов
        if (X[i] * Y[i] < A) { // Проверяем, меньше ли произведение X[i] и Y[i] значения A
            count++; // Увеличиваем счетчик, если условие выполнено
            productSum += X[i] * Y[i]; // Добавляем произведение к сумме
        }
    }

    printf("\nКоличество произведений меньше A: %d\n", count); // Выводим количество произведений, удовлетворяющих условию
    printf("Сумма таких произведений: %.2f\n", productSum); // Выводим сумму произведений, удовлетворяющих условию

    return 0; // Завершаем программу с возвращением 0
}

```
> [!IMPORTANT]
> Если Вы используете не Windows, а, например, macOS, то проблем с кодировкой быть не должно — в таком случае игнорируйте следующие строки:
> ```c
> ...
> #include <windows.h> // Подключение модуля windows.h для установки кодировки вывода
> ...
>      SetConsoleOutputCP(CP_UTF8); // Устанавливаем кодировку UTF-8 для вывода в консоли русских символов: иначе будут иероглифы
> ```
>
> В операционных системах семейства Linux могут быть свои заморочки с кодировкой. Если что-то работает не так, как ожидается, то заглядывайте в Google :)

Сложность задачи состоит в том, что мы впервые сталкиваемся с функцией генерации случайных чисел.
